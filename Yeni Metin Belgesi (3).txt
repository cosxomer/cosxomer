        function getStartOfWeek(date) {
            const day = date.getDay();
            const diff = date.getDate() - day + (day === 0 ? -6 : 1); 
            const startOfWeek = new Date(date);
            startOfWeek.setDate(diff);
            startOfWeek.setHours(0, 0, 0, 0); 
            return startOfWeek;
        }

        function calculateWeekStart(offset) {
            const today = new Date();
            const startOfWeek = getStartOfWeek(today);
            const targetDate = new Date(startOfWeek);
            targetDate.setDate(startOfWeek.getDate() + (offset * 7));
            return targetDate;
        }

        async function reorderTasksInStorage(dayIndex) {
            const taskList = document.querySelector(`#day-${dayIndex} .task-list`);
            if (!taskList) return;

            const reorderedTasks = [];
            const taskElements = taskList.querySelectorAll('.task-item');
            const existingTasks = await loadTasks(dayIndex);

            taskElements.forEach(item => {
                const taskText = item.getAttribute('data-task-text');
                const existingTask = existingTasks.find(t => t.text === taskText);
                if (existingTask) {
                    reorderedTasks.push(existingTask);
                }
            });
            await saveTasks(dayIndex, reorderedTasks);
        }

        async function addTask(dayIndex, text) {
            const inputText = text.trim();
            if (inputText === "") {
                showCustomAlert("G√∂rev adƒ± bo≈ü olamaz.", '#fbc02d');
                return;
            }

            let tasks = await loadTasks(dayIndex);
            if (tasks.some(t => t.text === inputText)) {
                showCustomAlert("Bu g√∂rev zaten bu g√ºnde var.", '#fbc02d');
                return;
            }

            tasks.push({ text: inputText, completed: false });
            await saveTasks(dayIndex, tasks);
            
            const taskList = document.querySelector(`#day-${dayIndex} .task-list`);
            if (taskList) {
                createTaskElement(taskList, dayIndex, inputText, false);
            }
            document.getElementById(`input-${dayIndex}`).value = '';
            showCustomAlert(`${DAY_NAMES[dayIndex]} g√ºn√ºne yeni g√∂rev eklendi.`, '#00bcd4');
        }

        async function deleteTaskFromStorage(dayIndex, text) {
            let tasks = await loadTasks(dayIndex);
            tasks = tasks.filter(t => t.text !== text);
            await saveTasks(dayIndex, tasks);
            showCustomAlert(`'${text}' g√∂revi silindi.`, '#ef4444');
        }

        async function updateTaskCompletion(dayIndex, text, completed) {
            let tasks = await loadTasks(dayIndex);
            const task = tasks.find(t => t.text === text);
            if (task) {
                task.completed = completed;
                await saveTasks(dayIndex, tasks);
                if(completed) {
                    showCustomAlert(`'${text}' tamamlandƒ±. Aferin √ñmer!`, '#10b981');
                } else {
                    showCustomAlert(`'${text}' tekrar a√ßƒ±ldƒ±.`, '#fbc02d');
                }
            }
        }

        async function moveTaskToNextDay(dayIndex, text) {
            const nextDayIndex = (dayIndex + 1) % 7;
            let currentTasks = await loadTasks(dayIndex);
            let nextDayTasks = await loadTasks(nextDayIndex);
            
            const taskToMove = currentTasks.find(t => t.text === text);
            if (!taskToMove) return;

            if (nextDayTasks.some(t => t.text === text)) {
                showCustomAlert("Bu g√∂rev zaten ertesi g√ºne eklenmi≈ü.", '#fbc02d');
                return;
            }

            currentTasks = currentTasks.filter(t => t.text !== text);
            nextDayTasks.push({ text: taskToMove.text, completed: false });

            await saveTasks(dayIndex, currentTasks);
            await saveTasks(nextDayIndex, nextDayTasks);

            await renderTasks(dayIndex);
            await renderTasks(nextDayIndex);
            showCustomAlert(`${text} g√∂revi ${DAY_NAMES[nextDayIndex]} g√ºn√ºne ta≈üƒ±ndƒ±.`, '#00bcd4');
        }

        function getDragAfterElement(container, y) {
            const draggableElements = [...container.querySelectorAll('.task-item:not(.dragging)')];

            return draggableElements.reduce((closest, child) => {
                const box = child.getBoundingClientRect();
                const offset = y - box.top - box.height / 2;
                if (offset < 0 && offset > closest.offset) {
                    return { offset: offset, element: child };
                } else {
                    return closest;
                }
            }, { offset: Number.NEGATIVE_INFINITY }).element;
        }

        function handleDragOver(e) {
            e.preventDefault();
            const currentList = e.currentTarget;
            if (!draggedItem) return;
            const afterElement = getDragAfterElement(currentList, e.clientY);
            
            if (afterElement == null) {
                currentList.appendChild(draggedItem);
            } else {
                currentList.insertBefore(draggedItem, afterElement);
            }
        }

        async function handleDrop(e) {
            e.preventDefault();
            if (!draggedItem) return;

            const currentList = e.currentTarget;
            const currentDayIndex = parseInt(currentList.dataset.dayIndex);
            
            const originalTaskText = draggedItem.getAttribute('data-task-text');
            const originalDayIndex = draggedItem.closest('.day-cell').dataset.dayIndex;

            if (originalDayIndex != currentDayIndex) {
                // Farklƒ± g√ºnden geliyorsa ta≈üƒ±ma mantƒ±ƒüƒ±nƒ± kullan
                draggedItem.remove();
                // Eski listeden kaldƒ±r
                await deleteTaskFromStorage(parseInt(originalDayIndex), originalTaskText);
                // Yeni g√ºne ekle
                await addTask(currentDayIndex, originalTaskText);
                showCustomAlert(`${originalTaskText} g√∂revi ${DAY_NAMES[currentDayIndex]} g√ºn√ºne ta≈üƒ±ndƒ±.`, '#00bcd4');
                // renderTasks(originalDayIndex) zaten deleteTaskFromStorage i√ßinde yapƒ±lƒ±yor.
            } else {
                // Aynƒ± g√ºn i√ßinde sƒ±ralama deƒüi≈üti
                await reorderTasksInStorage(currentDayIndex);
            }
        }

        function setupDragAndDropListeners() {
            const taskLists = document.querySelectorAll('.task-list');
            taskLists.forEach(list => {
                list.addEventListener('dragover', handleDragOver);
                list.addEventListener('drop', handleDrop);
                list.addEventListener('dragenter', (e) => {
                    e.preventDefault();
                    list.classList.add('drag-over');
                });
                list.addEventListener('dragleave', () => {
                    list.classList.remove('drag-over');
                });
            });
        }

        function savePomodoroState() {
            const state = {
                isRunning: isTimerRunning,
                isPaused: isPaused,
                initialSeconds: initialSeconds,
                remainingTime: isTimerRunning ? expectedTime - Date.now() : remainingTimeOnPause,
                timestamp: Date.now()
            };
            localStorage.setItem('pomodoroState', JSON.stringify(state));
        }

        function loadPomodoroState() {
            const stateJson = localStorage.getItem('pomodoroState');
            if (!stateJson) return;

            const state = JSON.parse(stateJson);
            initialSeconds = state.initialSeconds;
            
            const hours = Math.floor(initialSeconds / 3600);
            const minutes = Math.floor((initialSeconds % 3600) / 60);
            const seconds = initialSeconds % 60;
            
            const studyHours = document.getElementById('study-hours');
            if(studyHours) studyHours.value = hours;
            const studyMinutes = document.getElementById('study-minutes');
            if(studyMinutes) studyMinutes.value = minutes;
            const studySeconds = document.getElementById('study-seconds');
            if(studySeconds) studySeconds.value = seconds;

            if (state.isRunning && !state.isPaused) {
                const elapsedSinceSave = Date.now() - state.timestamp;
                remainingTimeOnPause = state.remainingTime - elapsedSinceSave;
                
                if (remainingTimeOnPause <= 1000) {
                    resetTimer(true, true);
                    finishTimer(true);
                    return;
                }
                isTimerRunning = true;
                isPaused = false;
                expectedTime = Date.now() + remainingTimeOnPause;
                startInterval();
                disableTimerInputs(true);
            } else {
                isTimerRunning = state.isRunning;
                isPaused = state.isPaused;
                remainingTimeOnPause = state.remainingTime;
                updateTimerDisplay(remainingTimeOnPause);
            }
        }

        function startInterval() {
            if(timerInterval) clearInterval(timerInterval);
            timerInterval = setInterval(() => updateTimerDisplay(null, true), 100);
        }

        function updateTimerFromInputs() {
            const hours = getInputValue('study-hours');
            const minutes = getInputValue('study-minutes');
            const seconds = getInputValue('study-seconds');
            
            const safeHours = Math.max(0, Math.min(99, hours));
            const safeMinutes = Math.max(0, Math.min(59, minutes));
            const safeSeconds = Math.max(0, Math.min(59, seconds));
            
            document.getElementById('study-hours').value = safeHours;
            document.getElementById('study-minutes').value = safeMinutes;
            document.getElementById('study-seconds').value = safeSeconds;
            
            const newInitialSeconds = (safeHours * 3600) + (safeMinutes * 60) + safeSeconds;
            initialSeconds = Math.max(0, newInitialSeconds);
        }

        function updateTimerFromInputsAndReset() {
            if (!isTimerRunning || isPaused) {
                updateTimerFromInputs();
                resetTimer(true, false);
            }
            savePomodoroState();
        }

        function formatTime(totalSeconds) {
            totalSeconds = Math.max(0, totalSeconds || 0);
            const hours = Math.floor(totalSeconds / 3600);
            const minutes = Math.floor((totalSeconds % 3600) / 60);
            const seconds = Math.floor(totalSeconds % 60);
            const milliseconds = Math.round((totalSeconds - Math.floor(totalSeconds)) * 1000);
            
            const paddedHours = hours.toString().padStart(2, '0');
            const paddedMinutes = minutes.toString().padStart(2, '0');
            const paddedSeconds = seconds.toString().padStart(2, '0');
            const paddedMs = Math.floor(milliseconds / 100).toString(); // Y√ºzde bir saniye
            
            return `${paddedHours}:${paddedMinutes}:${paddedSeconds}`; 
            // Milisaniyeyi g√∂sterirsen hƒ±zlƒ± g√ºncelleme i√ßin: return `${paddedHours}:${paddedMinutes}:${paddedSeconds}.${paddedMs}`;
        }

        function updateTimerDisplay(remainingMs = null, checkExpiry = false) {
            const now = Date.now();
            if (remainingMs === null) {
                if (isPaused) {
                    remainingMs = remainingTimeOnPause;
                } else {
                    remainingMs = expectedTime - now;
                }
            }
            
            const totalSeconds = remainingMs / 1000;
            const timerDisplay = document.getElementById('timer-display');
            if (timerDisplay) {
                timerDisplay.textContent = formatTime(totalSeconds);
            }

            if (checkExpiry && remainingMs <= 0) {
                clearInterval(timerInterval);
                finishTimer(true);
            }
            
            if (isTimerRunning && !isPaused) {
                remainingTimeOnPause = remainingMs; // √áalƒ±≈üƒ±rken kalan s√ºreyi s√ºrekli g√ºncelle
            }

            savePomodoroState();
        }

        function startTimer() {
            if (!isTimerRunning) {
                updateTimerFromInputs(); // Ba≈ülatmadan √∂nce inputlardan al
                expectedTime = Date.now() + (initialSeconds * 1000);
                isTimerRunning = true;
                isPaused = false;
                startInterval();
                disableTimerInputs(true);
                const controlBtn = document.getElementById('timer-control-btn');
                if(controlBtn) {
                    controlBtn.textContent = "DURAKLAT";
                    controlBtn.onclick = pauseTimer;
                }
                // M√ºzik kontrol√º
                const musicSelect = document.getElementById('music-select');
                if(musicSelect.value !== 'none' && audioPlayer.paused) {
                    toggleMusic(document.getElementById('music-toggle-btn'));
                }
            } else if (isPaused) {
                expectedTime = Date.now() + remainingTimeOnPause;
                isPaused = false;
                startInterval();
                const controlBtn = document.getElementById('timer-control-btn');
                if(controlBtn) {
                    controlBtn.textContent = "DURAKLAT";
                    controlBtn.onclick = pauseTimer;
                }
                if(audioPlayer && audioPlayer.paused) {
                    toggleMusic(document.getElementById('music-toggle-btn'));
                }
            }
            savePomodoroState();
        }

        function pauseTimer() {
            if (isTimerRunning && !isPaused) {
                clearInterval(timerInterval);
                isPaused = true;
                // remainingTimeOnPause deƒüeri en son updateTimerDisplay i√ßinde ayarlanmƒ±≈ü olmalƒ±
                const controlBtn = document.getElementById('timer-control-btn');
                if(controlBtn) {
                    controlBtn.textContent = "DEVAM ET";
                    controlBtn.onclick = startTimer;
                }
                if(audioPlayer && !audioPlayer.paused) {
                    toggleMusic(document.getElementById('music-toggle-btn'));
                }
            }
            savePomodoroState();
        }

        async function finishTimer(isCompleted) {
            if (timerInterval) clearInterval(timerInterval);
            if (audioPlayer) audioPlayer.pause();
            
            const initialMs = initialSeconds * 1000;
            const remainingMs = Math.max(0, remainingTimeOnPause);
            const workedTimeMs = initialMs - remainingMs;
            
            const workedMinutes = Math.round(workedTimeMs / 60000); // Yuvarlanmƒ±≈ü dakika

            isTimerRunning = false;
            isPaused = false;

            if (workedMinutes > 0) {
                const dayIndex = (new Date().getDay() + 6) % 7;
                const dayName = DAY_NAMES[dayIndex];
                
                await saveStudyScore(dayIndex, workedMinutes);
                
                const formattedScore = formatMinutesToHoursMinutes(workedMinutes);
                let message = isCompleted ? 
                    `Tebrikler, √ñmer! ${formattedScore} √ßalƒ±≈üma ba≈üarƒ±yla tamamlandƒ± ve ${dayName} g√ºn√ºne kaydedildi.` : 
                    `√ñmer, ${formattedScore} √ßalƒ±≈üma ${dayName} g√ºn√ºne kaydedildi.`;
                showCustomAlert(message, '#38bdf8');
            } else if (!isCompleted && initialSeconds > 0) {
                showCustomAlert("√áok kƒ±sa √ßalƒ±≈ütƒ±nƒ±z, s√ºre kaydedilmedi.");
            }
            
            resetTimer(true, true);
            hidePomodoroModal();
            localStorage.removeItem('pomodoroState');
        }

        function resetTimer(clearIntervalFlag = true, enableInputs = true) {
            if (clearIntervalFlag) clearInterval(timerInterval);
            isTimerRunning = false;
            isPaused = false;
            
            // Input deƒüerlerini alƒ±p initialSeconds'ƒ± yeniden hesapla
            updateTimerFromInputs(); 
            remainingTimeOnPause = initialSeconds * 1000;

            updateTimerDisplay(remainingTimeOnPause);
            disableTimerInputs(!enableInputs); // reset yaparken inputlar a√ßƒ±k kalabilir

            const controlBtn = document.getElementById('timer-control-btn');
            if(controlBtn) {
                controlBtn.textContent = "BA≈ûLAT";
                controlBtn.onclick = startTimer;
            }
            if(audioPlayer && !audioPlayer.paused) {
                 toggleMusic(document.getElementById('music-toggle-btn'));
            }
        }

        function getNextYksDate(day, month, hour, minute) {
            const today = new Date();
            let year = today.getFullYear();
            
            // Javascript'te aylar 0'dan ba≈ülar (0 = Ocak, 5 = Haziran)

            if (targetDate <= today) {
                targetDate = new Date(year + 1, month, day, hour, minute, 0);
            }
            return targetDate;
        }

        function updateYksCountdown() {
            const now = new Date();
            const distance = YKS_TYT_DATE.getTime() - now.getTime();

            if (distance < 0) {
                clearInterval(yksInterval);
                const container = document.getElementById('yks-countdown-container');
                if (container) {
                    container.innerHTML = `<h2 style="color: var(--pomodoro-accent); margin: auto;">YKS GER√áEKLE≈ûTƒ∞! YENƒ∞ HEDEFLER BELƒ∞RLEME ZAMANI.</h2>`;
                    container.style.justifyContent = 'center';
                }
                return;
            }

            const totalDays = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hoursLeft = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutesLeft = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const secondsLeft = Math.floor((distance % (1000 * 60)) / 1000);

            // Deƒüerleri G√ºncelle
            const daysValue = document.getElementById('days-value');
            if (daysValue) daysValue.textContent = totalDays;
            const hoursValue = document.getElementById('hours-value');
            if (hoursValue) hoursValue.textContent = hoursLeft.toString().padStart(2, '0');
            const minutesValue = document.getElementById('minutes-value');
            if (minutesValue) minutesValue.textContent = minutesLeft.toString().padStart(2, '0');
            const secondsValue = document.getElementById('seconds-value');
            if (secondsValue) secondsValue.textContent = secondsLeft.toString().padStart(2, '0');

            // √áemberleri G√ºncelle
            const secondsRatio = secondsLeft / 60;
            setDashOffset('seconds-circle', secondsRatio);
            const minutesRatio = (minutesLeft + secondsRatio) / 60;
            setDashOffset('minutes-circle', minutesRatio);
            const hoursRatio = (hoursLeft + minutesRatio * 60 / 3600) / 24; 
            setDashOffset('hours-circle', hoursRatio); 
            const daysRatio = totalDays % 365 / 365; // Yƒ±llƒ±k g√∂sterim
            setDashOffset('days-circle', daysRatio);
        }

        function setTheme(themeName, button) {
            const theme = themes[themeName];
            if (!theme) return;
            
            for (const [key, value] of Object.entries(theme)) {
                document.documentElement.style.setProperty(key, value);
            }
            
            document.body.className = themeName === 'default' ? '' : `${themeName}-mode`;
            localStorage.setItem('theme', themeName);
            
            document.querySelectorAll('.theme-button').forEach(btn => btn.classList.remove('active'));
            if (button) button.classList.add('active');
        }

        function loadTheme() {
            const savedTheme = localStorage.getItem('theme') || 'default';
            const button = document.getElementById(`${savedTheme}-btn`);
            setTheme(savedTheme, button);
        }

        function formatMinutesToHoursMinutes(totalMinutes) {
            if (totalMinutes < 1) return `0 dk`;
            if (totalMinutes < 60) return `${totalMinutes} dk`;

            const hours = Math.floor(totalMinutes / 60);
            const minutes = Math.round(totalMinutes % 60);
            
            let result = '';
            if (hours > 0) {
                result += `${hours} sa`;
            }
            if (minutes > 0 || hours === 0) {
                if (hours > 0) result += ' ';
                result += `${minutes} dk`;
            }
            return result.trim();
        }

        async function updateWeeklyTotalTime() {
            const scores = await loadStudyScores();
            const totalMinutes = scores.reduce((sum, score) => sum + (score || 0), 0);
            const displayElement = document.getElementById('weekly-total-time');
            if (displayElement) {
                displayElement.textContent = `HAFTA: ${formatMinutesToHoursMinutes(totalMinutes)}`;
            }
            updateTodayWorkedTime(scores);
        }
        
        async function renderStudyScores() {
            const scores = await loadStudyScores();
            let todayWorkedMinutes = 0;
            const currentDayIndex = (new Date().getDay() + 6) % 7; 

            for (let i = 0; i < 7; i++) {
                const scoreMinutes = scores[i] || 0;
                const scoreArea = document.getElementById(`score-${i}`);
                
                if (i === currentDayIndex) {
                    todayWorkedMinutes = scoreMinutes;
                }
                
                if (scoreArea) {
                    scoreArea.textContent = `√áalƒ±≈üma: ${formatMinutesToHoursMinutes(scoreMinutes)}`;
                }
            }
            
            const todayWorkedDisplay = document.getElementById('today-worked-time');
            if (todayWorkedDisplay) {
                todayWorkedDisplay.textContent = `BUG√úN: ${formatMinutesToHoursMinutes(todayWorkedMinutes)}`;
            }
        }
        
        function updateTodayWorkedTime(scores = null) {
            // Bu fonksiyon renderStudyScores tarafƒ±ndan √ßaƒürƒ±lƒ±yor ama burada tutarlƒ±lƒ±k i√ßin g√ºncelledik
            if (!scores) return; // Eƒüer skorlar renderStudyScores'dan gelmezse hesaplama

            let todayWorkedMinutes = 0;
            const currentDayIndex = (new Date().getDay() + 6) % 7; 

            if(scores[currentDayIndex]) {
                todayWorkedMinutes = scores[currentDayIndex];
            }
            
            const todayWorkedDisplay = document.getElementById('today-worked-time');
            if (todayWorkedDisplay) {
                todayWorkedDisplay.textContent = `BUG√úN: ${formatMinutesToHoursMinutes(todayWorkedMinutes)}`;
            }
        }

        function changeMusicVolume(volume) {
            musicVolume = parseInt(volume);
            localStorage.setItem('musicVolume', musicVolume);
            if (audioPlayer) audioPlayer.volume = musicVolume / 100;
        }

        function toggleMusic(button) {
            if (!audioPlayer) return;
            if (audioPlayer.paused) {
                audioPlayer.play();
                button.innerHTML = '<i class="fas fa-pause"></i> M√ºzik Duraklat';
            } else {
                audioPlayer.pause();
                button.innerHTML = '<i class="fas fa-play"></i> M√ºzik Ba≈ülat';
            }
        }

        function changeMusicSource(source) {
            if (!audioPlayer) return;
            if (source === 'none') {
                audioPlayer.pause();
                audioPlayer.removeAttribute('src');
            } else {
                audioPlayer.src = source;
                audioPlayer.play();
            }
            const toggleBtn = document.getElementById('music-toggle-btn');
            if(toggleBtn) toggleBtn.innerHTML = '<i class="fas fa-pause"></i> M√ºzik Duraklat';
        }

        }

        function showCustomAlert(message, backgroundColor = '#ef4444') {
            const alertBox = document.getElementById('custom-alert');
            if (!alertBox) return;
            
            alertBox.style.backgroundColor = backgroundColor;
            alertBox.textContent = message;
            alertBox.style.opacity = '1';
            alertBox.style.top = '30px'; 

            setTimeout(() => {
                alertBox.style.opacity = '0';
                alertBox.style.top = '-50px';
